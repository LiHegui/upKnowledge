    constructor(props) {
        this.messages = {}
        this.listeners = {}
    }

    // æ·»åŠ å‘å¸ƒè€…
    publish(type, content) {
        const existContent = this.messages[type]
        if (!existContent) {
            this.messages[type] = [];
        }
        this.messages[type].push(content)
    }

    // æ·»åŠ è®¢é˜…è€…
    subscribe(type, cb) {
        const existListener = this.listeners[type]
        if (!existListener) {
            this.listeners[type] = []
        }
        this.listeners[type].push(cb)
    }

    // é€šçŸ¥
    notify(type) {

    }
}
